# Complexity: O(n), O(n)
def solve(s):
    filtered_s = []
    for char in s:
        if char.isalnum():
            filtered_s.append(char.lower())

    len_s = len(filtered_s)
    if len_s == 0:
        return True

    i, j = 0, len_s -1
    while i < j:
        if filtered_s[i] != filtered_s[j]:
            return False
        i += 1
        j -= 1
    return True
