# Complexity: O(n), O(1)
def solve(nums):
    zero_idx = -1
    for i in range(len(nums)):
        if nums[i] == 0 and zero_idx == -1:
            zero_idx = i
        if nums[i] != 0 and zero_idx != -1:
            nums[i], nums[zero_idx] = nums[zero_idx], nums[i]
            zero_idx += 1
    return nums
